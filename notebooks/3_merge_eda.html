Between-person convergence

Verbal

  • verbb_init: I always ask verbally for consent before I initiate a sexual encounter
  • verbb_ong: I always ask verbally for consent at each step of a sexual encounter
verbal_means <- d %>%
  group_by(nid) %>%
  summarise(
    mean_ver_pc = mean(ver_pc, na.rm = TRUE),
    n_obs = sum(!is.na(ver_pc))
  ) %>%
  ungroup()

d <- d %>%
  left_join(verbal_means, by = "nid")

round(cor(d$verbb_init, d$mean_ver_pc, use = "pairwise.complete.obs"), 2)
[1] 0.53
