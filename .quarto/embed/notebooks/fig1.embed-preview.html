d <- d %>%
  mutate(
    verb_bins = cut(
      ver_pc,
      breaks = c(-Inf, 0, 3, 6, 9, Inf),
      labels = c("Not at all", "A little", "Somewhat", "Quite a bit", "Very much"),
      right = TRUE,
      include.lowest = TRUE
    ),
    verb_bins = ordered(verb_bins,
                        levels = c("Not at all", "A little", "Somewhat", "Quite a bit", "Very much"))
  )

# Verify
is.ordered(d$verb_bins)
